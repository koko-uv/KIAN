أدوات الأتمتة الأمنية (السكربتات)

مجموعة من السكربتات المتقدمة لتأمين الأنظمة، فحص الثغرات، وتحليل السجلات. هذه الأدوات مصممة للمستخدمين ذوي المعرفة التقنية العالية.

سكربت بايثون: فحص سلامة الملفات (MD5 Hash Checker)

يقوم هذا السكربت بحساب قيمة تجزئة MD5 لملف معين، مما يساعد في التحقق من أن الملف لم يتم التلاعب به أو تعديله بعد تنزيله أو نقله. قم بنسخ الكود ولصقه في ملف `check_file_integrity.py` ثم شغله باستخدام `python check_file_integrity.py`.

# -*- coding: utf-8 -*-
# سكربت فحص سلامة الملفات باستخدام MD5 Hash
# يستخدم للتحقق من أن الملف لم يتم تعديله

import hashlib
import os

def calculate_md5(file_path, chunk_size=8192):
    """
    يحسب قيمة تجزئة MD5 لملف معين.
    :param file_path: مسار الملف.
    :param chunk_size: حجم القطعة للقراءة (للملفات الكبيرة).
    :return: قيمة MD5 كـسلسلة نصية سداسية عشرية، أو None إذا كان هناك خطأ.
    """
    md5_hash = hashlib.md5()
    if not os.path.exists(file_path):
        print(f"خطأ: الملف غير موجود في المسار: {file_path}")
        return None
    
    try:
        with open(file_path, 'rb') as f:
            for chunk in iter(lambda: f.read(chunk_size), b''):
                md5_hash.update(chunk)
        return md5_hash.hexdigest()
    except Exception as e:
        print(f"حدث خطأ أثناء قراءة الملف: {e}")
        return None

def main():
    file_to_check = input("الرجاء إدخال مسار الملف للتحقق من سلامته: ")
    expected_md5 = input("الرجاء إدخال قيمة MD5 المتوقعة (إذا كانت لديك): ").strip()

    calculated_md5 = calculate_md5(file_to_check)

    if calculated_md5:
        print(f"\nقيمة MD5 المحسوبة للملف: {calculated_md5}")
        if expected_md5:
            if calculated_md5 == expected_md5:
                print("تطابق القيم: سلامة الملف مؤكدة! ✅")
            else:
                print("عدم تطابق القيم: قد يكون الملف قد تعرض للتعديل! ❌")
        else:
            print("لم يتم توفير قيمة MD5 متوقعة للمقارنة.")
    else:
        print("لم نتمكن من حساب قيمة MD5 للملف.")

if __name__ == "__main__":
    main()

**ملاحظة هامة:** هذا السكربت هو مثال توضيحي. لضمان أمان وفعالية أي سكربتات، تأكد دائمًا من فهمك الكامل لعملها، وقم بتنزيلها من مصادر موثوقة، وافحصها جيدًا قبل التنفيذ.
